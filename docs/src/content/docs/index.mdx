## Code Transformation

See how Python transforms to clean TypeScript:

<Tabs>
  <TabItem label="Functions & Types">
```python
# Python
def greet(name: str, times: int = 1) -> str:
    return f"Hello, {name}!" * times
```

```typescript
// TypeScript
function greet(name: string, times: number = 1): string {
  return `Hello, ${name}!`.repeat(times)
}
```
  </TabItem>
  <TabItem label="Classes & Dataclasses">
```python
# Python
@dataclass
class Point:
    x: float
    y: float
    label: str = "origin"
```

```typescript
// TypeScript
class Point {
  constructor(
    public x: number,
    public y: number,
    public label: string = "origin"
  ) {}
}
```
  </TabItem>
  <TabItem label="Generics">
```python
# Python
T = TypeVar('T')

class Container(Generic[T]):
    def __init__(self, value: T) -> None:
        self.value = value
```

```typescript
// TypeScript
class Container<T> {
  value: T

  constructor(value: T) {
    this.value = value
  }
}
```
  </TabItem>
  <TabItem label="Comprehensions">
```python
# Python
squares = [x ** 2 for x in range(10)]
evens = [x for x in nums if x % 2 == 0]
lookup = {k: v for k, v in pairs}
```

```typescript
// TypeScript
let squares = range(10).map(x => pow(x, 2))
let evens = nums.filter(x => mod(x, 2) === 0)
let lookup = Object.fromEntries(pairs)
```
  </TabItem>
</Tabs>

## pythonlib Modules

Use Python's best APIs in TypeScript — standalone, without the transpiler:

<CardGrid>
  <Card title="itertools" icon="list-format">
    `chain`, `combinations`, `permutations`, `product`, `cycle`, `groupby`, `takeWhile`, `dropWhile`

    ```typescript
    import { combinations } from "pythonlib/itertools"
    [...combinations([1, 2, 3], 2)]
    // [[1,2], [1,3], [2,3]]
    ```
  </Card>
  <Card title="functools" icon="setting">
    `lruCache`, `partial`, `reduce`, `pipe`, `compose`, `cache`

    ```typescript
    import { lruCache } from "pythonlib/functools"
    const fib = lruCache((n: number): number =>
      n <= 1 ? n : fib(n - 1) + fib(n - 2)
    )
    ```
  </Card>
  <Card title="collections" icon="puzzle">
    `Counter`, `defaultdict`, `deque`, `OrderedDict`

    ```typescript
    import { Counter } from "pythonlib/collections"
    new Counter("mississippi").mostCommon(2)
    // [["i", 4], ["s", 4]]
    ```
  </Card>
  <Card title="datetime" icon="seti:clock">
    `datetime`, `date`, `time`, `timedelta`

    ```typescript
    import { datetime, timedelta } from "pythonlib/datetime"
    datetime.now().add(timedelta({ days: 7 }))
    ```
  </Card>
  <Card title="pathlib" icon="document">
    `Path` with `exists`, `readText`, `writeText`, `glob`

    ```typescript
    import { Path } from "pythonlib/pathlib"
    const p = new Path("/home/user/file.txt")
    p.stem // "file"
    ```
  </Card>
  <Card title="hashlib" icon="seti:lock">
    `md5`, `sha1`, `sha256`, `sha512`, `blake2b`

    ```typescript
    import { sha256 } from "pythonlib/hashlib"
    sha256("password").hexdigest()
    ```
  </Card>
  <Card title="re" icon="magnifier">
    `search`, `match`, `findAll`, `sub`, `compile` with named groups

    ```typescript
    import { search } from "pythonlib/re"
    search("(?P<name>\\w+)@(?P<domain>\\w+)", email)
    ```
  </Card>
  <Card title="random" icon="random">
    `randInt`, `choice`, `shuffle`, `sample`, `uniform`

    ```typescript
    import { choice, shuffle } from "pythonlib/random"
    choice(["a", "b", "c"])
    ```
  </Card>
</CardGrid>

## Quick Start

<Tabs>
  <TabItem label="Transpiler">
```bash
# Install
npm install python2ts

# Transpile a file
npx python2ts input.py -o output.ts

# Or use programmatically
import { transpile } from "python2ts"
const ts = transpile(pythonCode)
```
  </TabItem>
  <TabItem label="Runtime Only">
```bash
# Install just the runtime
npm install pythonlib
```

```typescript
// Use Python-style APIs directly
import { range, enumerate, zip } from "pythonlib"
import { combinations } from "pythonlib/itertools"
import { Counter } from "pythonlib/collections"
import { Path } from "pythonlib/pathlib"

for (const [i, item] of enumerate(items)) {
  console.log(i, item)
}
```
  </TabItem>
</Tabs>
